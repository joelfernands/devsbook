git remote add origin https://github.com/joelfernands/devsbook.git
git branch -M devsbook

git add -A
git commit -m "Iniciando no git"

git status                              ver status
git log                                 ver comits 
git add -A                              traquear arquivos
git commit -m "<descrição>"             descrição da alteração 
git branch <nome>                       ver branchs *é o current
git checkout <branch> ou <arquivo>      mudar de branch 
git reset --soft <codigo comit>         Desfaz o comit mas não exclui o arquivo  
git reset --mixed <codigo comit>        Desfaz o comit mas não exclui o arquivo e volta para o status antes do (git add -A)
git reset --hard <codigo comit>         Exclui o comite
git revert  --no-edit <codigo commit>   Remove alterações mais não exclui o commite anterior

git diff                                para ver o que foi alterado

git diff <nome.extensão>                Ver somente deste arquivo
git checkout HEAD -- <nome.extensão>    Excluindo alterações


.gitignore                              Mantem os arquivos mencionados dentro dele, pode ser pelo nome. ext ou todos com a *.

git pull origin master                  atualizando repositorio local

git push origin master                  Atualizando repositorio remoto
